package framework.util;

import java.beans.PropertyDescriptor;
import java.net.URL;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collection;
import java.util.Iterator;
import java.util.Map;
import org.apache.commons.beanutils.PropertyUtils;
import org.apache.commons.logging.Log;
import org.apache.commons.logging.LogFactory;

public final class ObjectUtils {
    private static Log log = LogFactory.getLog(ObjectUtils.class);

    protected ObjectUtils() {
    }

    public static Collection toCollection(Object value) {
        if (value == null) {
            return null;
        } else {
            Collection result = new ArrayList();
            if (value instanceof Collection) {
                if (log.isDebugEnabled()) {
                    log.debug("toCollection: Value is a already a Collection");
                }

                result.addAll((Collection)value);
            } else if (value.getClass().isArray()) {
                if (log.isDebugEnabled()) {
                    log.debug("toCollection: Value is an array");
                }

                result.addAll(Arrays.asList((Object[])((Object[])value)));
            } else {
                if (log.isDebugEnabled()) {
                    log.debug("toCollection: Value is considered as a single value");
                }

                result.add(value);
            }

            return result;
        }
    }
}
